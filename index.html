      success: false,
      message: 'Erreur de base de donn√©es'
    });
  }

  // Erreur par d√©faut
  res.status(err.status || 500).json({
    success: false,
    message: err.message || 'Erreur interne du serveur',
    ...(process.env.NODE_ENV === 'development' && { stack: err.stack })
